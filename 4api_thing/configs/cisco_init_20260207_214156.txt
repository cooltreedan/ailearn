!

! === BASIC SYSTEM CONFIGURATION ===
! Set device hostname
hostname homelab-r1

! Enable password encryption for all passwords
service password-encryption

! Enable timestamps on log messages for troubleshooting
service timestamps debug datetime msec localtime show-timezone
service timestamps log datetime msec localtime show-timezone

! Disable unnecessary services for security
no service pad
no service finger
no service udp-small-servers
no service tcp-small-servers
no ip http server
no ip http secure-server
no cdp run

! Set timezone (modify according to your location)
! Example: clock timezone EST -5 0
! Example: clock summer-time EDT recurring

! === SECURITY CONFIGURATION ===
! Set enable secret (CHANGE THIS PASSWORD!)
enable secret Changeme123!

! Configure console access
line con 0
 password Console123!
 login
 logging synchronous
 exec-timeout 15 0
 transport preferred none

! Configure VTY (Telnet/SSH) access
line vty 0 4
 password VTY123!
 login local
 logging synchronous
 exec-timeout 15 0
 transport input ssh
 transport output ssh

! Configure auxiliary port
line aux 0

 login

 transport input none
 transport output none

! Create local admin user (CHANGE USERNAME AND PASSWORD!)
username admin privilege 15 secret Admin123!

ip domain-name homelab.local

! Generate RSA key for SSH (may take a moment)
crypto key generate rsa general-keys modulus 2048

! Configure SSH version 2 only
ip ssh version 2
ip ssh time-out 60
ip ssh authentication-retries 3

! === LOGGING CONFIGURATION ===
! Configure logging
logging buffered 16384
logging console warnings
logging monitor warnings

! logging host 192.168.1.100
